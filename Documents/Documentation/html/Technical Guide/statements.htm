1. if-else statements :
            
if (condition)
{
      statements;
}
else if (condition)
{
      statements;
}
else if (condition)
{
      statements;
}
                        
2. for statements :

for (initialization; condition; update)
{
      statements;
}
                        
3. do-while Statements :
do
{
      statements;
} while (condition);
                        
4. switch statements :
            
            switch (condition)
      {
            case ABC:
            statements;
            /* falls through */
            case DEF:
                  statements;
            break;
            case XYZ:
                  statements;
            break;
            default:
                  statements;
                        break;
      }
            
5. try-catch Statements :
            
            try
      {
            statements;
      }     catch (ExceptionClass e)
            {
                  statements;
            }
